<script src="//code.jquery.com/jquery-1.7.2.min.js"></script>
<script src="js/ygrab.js"></script>
<script>		
$(function() {
	
	var data = {
		url: 'https://bachors.js.org/',
		selector: 'div.project-card',
		loop: true,
		result: [
			{
				name: 'title',
				find: 'h3',
				grab: {
					by: 'text',
					value: ''
				}
			},
			{
				name: 'image',
				find: 'img',
				grab: {
					by: 'attr',
					value: 'src'
				}
			},
			{
				name: 'link',
				find: 'a.text-secondary-light',
				grab: {
					by: 'attr',
					value: 'href'
				}
			}
		]
	};
	
	ygrab(data, function(result) {
		$('#res').html(JSON.stringify(result, null, 2));
	});
	
});
</script>
